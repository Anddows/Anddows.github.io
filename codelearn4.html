import telebot

bot = telebot.TeleBot('botingiz tokeni')

#start bosganingizda sizga javob qayarishi uchun:

@bot.message_handler(commands = ['start'])
def start_message(message):
  bot.reply_to(message, "salom")

@bot.message_handler(content_types = ['text'])
def send_message(message):

  if message.text.lower() == "salom":
    bot.send_message(message.chat.id, "salom code idea ga hush kelibsiz")

  elif message.text.lower() == "rahmat":
    bot.send_message(message.chat.id, "hursand bo'lgan bo'lsangiz biz ham hursandmiz")

#if dan keyin elif funksiyasini qo'llashingizni maslahat beramiz.

#botimiz o'chib qolmasligi uchun biz:

bot.polling(none_stop = True)
